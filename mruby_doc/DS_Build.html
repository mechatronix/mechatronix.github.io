def generate_all_sources(base_path)
  return unless COMMON_Build.enabled?
  base_path += "/templates_DS/" ;
  lbase     = base_path.length
  Dir.find("#{base_path}", /\.tmpl$/) do |input_file_name|
    output_file_name = @outputDir + input_file_name.sub('MODEL',@modelName)[lbase..-6] # elimino templates e .tmpl
    input_file_name.gsub!(/\\+/,'/')
    input_file_name.gsub!(/\/+/,'/')
    output_file_name.gsub!(/\\+/,'/')
    output_file_name.gsub!(/\/+/,'/')
  
    unless @exclude.include? File.basename(output_file_name) then 
      dir  = File.dirname(output_file_name)
      dirs = dir.split("/")
      path = ""
      dirs.each do |d|
        path += "#{d}"
        Dir.mkdir(path) unless File::directory?(path)
        path += "/"
      end
      merb = MERB.new
      if @debug_mode then
        puts "input file name = #{input_file_name}"
        File.open(input_file_name, 'r') { |f| merb.template = f.read }
        puts "SOURCE\n\n"
        puts merb.source
        puts "EVAL #{input_file_name}\n\n"
        eval(merb.source)
        puts MERB.out
        puts "output file name = #{output_file_name}"
      else
        merb.convert( input_file_name, output_file_name ) ;
      end
    end
  end

  if Dir.exists? "user-code" then
    Dir.find("./user-code", /\.(c|h).*$/) do |input_file_name|
      input = File.open(input_file_name, 'r' ) { |f| f.read() }
      output_file_name = @outputDir + '/ds-cpp/srcs/' + input_file_name[12..-1] # elimino user-code
      File.unlink(output_file_name) if File.exist? output_file_name
      File.open(output_file_name, 'w') { |f| f.write(input)  }
    end
  end    
  #print "ALL DONE\n"
end

def read_parameters()

  @maxlenX   = COMMON_Build::DATA[:states]              . collect { |item| item.length() }.max ;
  @maxlenC   = COMMON_Build::DATA[:controls]            . collect { |item| item.length() }.max ;
  @maxlenMP  = COMMON_Build::DATA[:ModelParameters]     . collect { |item| item[:name].length() }.max ;
  @maxlenAUX = COMMON_Build::DATA[:AuxiliaryParameters] . collect { |item| item[:name].length() }.max ;
  @maxlenU   = COMMON_Build::DATA[:ConstraintU]         . collect { |item| item[:name].length() }.max ;
  @maxlen1D  = COMMON_Build::DATA[:Constraint1D]        . collect { |item| item[:name].length() }.max ;
  @maxlen2D  = COMMON_Build::DATA[:Constraint2D]        . collect { |item| item[:name].length() }.max ;

  @maxlenX   = 0 unless @maxlenX
  @maxlenC   = 0 unless @maxlenC
  @maxlenMP  = 0 unless @maxlenMP
  @maxlenAUX = 0 unless @maxlenAUX
  @maxlenU   = 0 unless @maxlenU
  @maxlen1D  = 0 unless @maxlen1D
  @maxlen2D  = 0 unless @maxlen2D

  # cerca le librerie da agganciare in automatico e rimuove eventuale prefisso
  @UserClassInstance = []
  @UserClassPointer  = []
  @userHeaders       = {}
  @userNamespaces    = {}
  @userDataFile      = {}
  @lenicl            = 0
  @lencl             = 0
  COMMON_Build::DATA[:UserFunctionsClassInstances].each { |item|

    @userHeaders[item[:header]]       = 0 ; # simulo un insieme usando le chiavi di una hash
    @userNamespaces[item[:namespace]] = 0 ;
    @userDataFile[item[:setup_file]]  = 0 if item[:setup_file].kind_of? String ;

    cl_full = item[:class]
    cl      = item[:class].sub(/(^[^#]+#)?/,'')  # rimuovo prefisso
    icl     = item[:instance]
    # controllo se la istanza è di tipo puntatore o meno
    # se è un puntatore allora vuole dire che viene definita dall'utente esternamente
    # e viene passato il puntatore, altrimenti è una classe contenuta nel modello.
    if icl[0..0] == "*" then
      icl_p = icl[1..-1]
      icl   = icl[2..2].downcase + icl[3..-1] # nome senza "*p" con prima lettera minuscola
      @UserClassPointer << { :class => cl, :instance => icl, :instance_p => icl_p, :class_full => cl_full }
    else
      @UserClassInstance << { :class => cl, :instance => icl, :class_full => cl_full }
    end
    @lencl  = cl.length  if  cl.length > @lencl
    @lenicl = icl.length if icl.length > @lenicl

  }
end

def subs_numbers( lines )
  out = ""
  lines.each do |line|
    line.gsub!( /unknown/,"result") # workaround per codegen
    line.gsub!( /\_\_\_dot\_\_\_/,".")
    line.gsub!( /\_\_\_arrow\_\_\_/," -> ")
    line.gsub!( /\_\_\_void_argument\_\_\_/," ")
    line.gsub!( /operator\_\_\_pars\_\_\_\_/,"D_")
    line.gsub!( /operator\_\_\_pars\_\_\_/,"operator()")
    line.gsub!( /\bX\[\d+\]/  ) { |s| m = /\d+/.match(s) ; "X[iX_#{COMMON_Build::DATA[:states][m[0].to_i]}]" }
    line.gsub!( /\bU\[\d+\]/  ) { |s| m = /\d+/.match(s) ; "U[iU_#{COMMON_Build::DATA[:controls][m[0].to_i]}]" }
    line.gsub!( /\bModelPars\[\d+\]/) { |s| m = /\d+/.match(s) ; "ModelPars[iM_#{COMMON_Build::DATA[:ModelParameters][m[0].to_i][:name]}]" }
    line.gsub!( /\bpow\(([^,]+), +2\)/ )   { |s| m = /\([^,]+/.match(s) ; "power2(#{m[0][1..-1]})" }
    line.gsub!( /\bpow\(([^,]+), +3\)/ )   { |s| m = /\([^,]+/.match(s) ; "power3(#{m[0][1..-1]})" }
    line.gsub!( /\bpow\(([^,]+), +0\.2e1\)/ )   { |s| m = /\([^,]+/.match(s) ; "power2(#{m[0][1..-1]})" }
    line.gsub!( /\bpow\(([^,]+), +0\.3e1\)/ )   { |s| m = /\([^,]+/.match(s) ; "power3(#{m[0][1..-1]})" }
    line.gsub!( /\bpow\(([^,]+), +\-0\.2e1\)/ ) { |s| m = /\([^,]+/.match(s) ; "rpower2(#{m[0][1..-1]})" }
    line.gsub!( /\bpow\(([^,]+), +\-0\.3e1\)/ ) { |s| m = /\([^,]+/.match(s) ; "rpower3(#{m[0][1..-1]})" }
    line.gsub!( /\b0\.1e1\b/, "1.0" )
    line.gsub!( /\b0\.2e1\b/, "2.0" )
    line.gsub!( /\b0\.3e1\b/, "3.0" )
    line.gsub!( /\b0\.4e1\b/, "4.0" )
    line.gsub!( /\b0\.5e1\b/, "5.0" )
    line.gsub!( /\b0\.6e1\b/, "6.0" )
    line.gsub!( /\b0\.7e1\b/, "7.0" )
    line.gsub!( /\b0\.8e1\b/, "8.0" )
    line.gsub!( /\b0\.9e1\b/, "9.0" )
    out += line+"\n"
  end
  return out
end

def using_headers( indent = "" )
  tmp  = indent + "// Mechatronix Headers\n"
  tmp += indent + "#include <MechatronixCore/MechatronixCore.hh>\n"
  tmp += indent + "#include <MechatronixAstro/MechatronixAstro.hh>\n"
  tmp += indent + "#include <MechatronixInterfaceMruby/MechatronixInterfaceMruby.hh>\n"
  tmp += indent + "#include <MechatronixInterfaceLua/MechatronixInterfaceLua.hh>\n"
  tmp += indent + "#include <MechatronixODE/MechatronixODE.hh>\n"
  if @userHeaders.length > 0 then
    tmp += indent + "// user headers\n"
    @userHeaders.each { |k,h| tmp += indent + k + "\n" }
  end
  return tmp ;
end

def using_namespaces( indent = "" )
  tmp  = indent + "using namespace std ;\n"
  tmp += indent + "// Mechatronix namespaces\n"
  tmp += indent + "using namespace MechatronixCore ;\n"
  tmp += indent + "using namespace MechatronixAstro ;\n"
  tmp += indent + "using namespace MechatronixInterfaceMruby ;\n"
  tmp += indent + "using namespace MechatronixInterfaceLua ;\n"
  tmp += "\n"
  if @userNamespaces.length > 0 then
    tmp += indent + "// user namespaces\n"
    @userNamespaces.keys.each { |ns| tmp += indent + "using namespace #{ns} ;\n" }
  end
  return tmp ;
end
