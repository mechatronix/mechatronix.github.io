Overview

Note:

DataGraph is only supported on Mac.

Class for plotting simulation results with DataGraph

Examples:

Simple usage

dg = Dgraph.new
dg.set_fields %w(a b c)
dg.plot "a", "b"
dg.plot "a", "c"
100.times do |i|
  a = i / 10.0
  b = Math::sin(a)
  c = Math::cos(a)
  dg << {a:a, b:b, c:c}
end
dg.add_field "z"
dg.add_column("z", [1,2,3,4,5])
dg.close

With block

Dgraph.plot do |dg
  dg.set_fields %w(a b c)
  dg.plot "a", "b"
  dg.plot "a", "c"
  100.times do |i|
    a = i / 10.0
    b = Math::sin(a)
    c = Math::cos(a)
    dg << {a:a, b:b, c:c}
  end
  dg.add_field "z"
  dg.add_column("z", [1,2,3,4,5])
end

def initialize(path = DGRAPH)
  @cmd = path
  @dg  = IO.popen(@cmd, "w")
  @fields = []
  @dg.puts "new file"
  @dg.puts "delete column x"
  @dg.puts "delete column y"
  @dg.puts "stream on"
  @dg.puts "delete command 4"
  if block_given? then
    yield @dg
    self.close
  end
end

def <<(hsh)
  raise ArgumentError, "Need a Hash" unless hsh.kind_of? Hash
  @dg.puts "append #{hsh.keys.join(',')}"
  @dg.puts hsh.values.join(',')
  @dg.puts "*end*"
end

def add_column(name, ary)
  raise ArgumentError, "Need an Array" unless ary.kind_of? Array
  @dg.puts "append #{name}"
  @dg.puts ary.join("\n")
  @dg.puts "*end*"
end
