def self.find(path, rx, &block)
  list = []
  self.foreach(path) do |e|
    name = "#{path}/#{e}"
    if (File.directory?(name) && ! /\.{1,2}$/.match(name) ) then
      list += self.find(name, rx, &block)
    end
    if rx.match e then
      block.call(name) if block
      list << name 
    end
  end
  return list
end
