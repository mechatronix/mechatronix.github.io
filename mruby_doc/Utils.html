+ (Hash) read_from_table(filename)

Read a tabulated file into a Hash of Arrays. Keys are the column names. The method assumes spaces for column separator. Empty lines are skipped. Lines begining with # are considered comments and skipped as well.

def self.read_from_table(filename)
  raise ArgumentError, "Cannot load #{filename}" unless File.exist? filename
  fields     = []
  fields_num = nil
  line_num   = 0
  table      = {}
  File.foreach(filename) do |line|
    line_num += 1
    next if line.match(/^\s*#/) # Skip comment lines
    next if line.match(/^\s*$/) # Skip empty lines
    fields = line.chomp.split
    if line.match(/^\s*[a-zA-Z]/) && table.keys.empty? then # Keys
      fields.each {|f| table[f] = []}
      fields_num = table.keys.length        
    else
      raise RuntimeError, "Wrong fields count at line #{line_num}" if fields.length != fields_num
      fields.each_with_index {|f,i| table[table.keys[i]] << f.to_f}
    end
  end
  return table
end
