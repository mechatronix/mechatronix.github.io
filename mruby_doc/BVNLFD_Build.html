def generate_all_sources(base_path)
  return unless COMMON_Build.enabled?
  base_path += "/templates_OCP/" ;
  lbase      = base_path.length
  Dir.find("#{base_path}", /\.tmpl$/) do |input_file_name|
    output_file_name = @outputDir + input_file_name.sub('MODEL',@modelName)[lbase..-6] # elimino templates e .tmpl
    input_file_name.gsub!(/\\+/,'/')
    input_file_name.gsub!(/\/+/,'/')
    output_file_name.gsub!(/\\+/,'/')
    output_file_name.gsub!(/\/+/,'/')

    unless @exclude.include?(File.basename(output_file_name)) then
      dir  = File.dirname(output_file_name)
      dirs = dir.split("/")
      path = ""
      dirs.each do |d|
        #puts path
        path += "#{d}"
        Dir.mkdir(path) unless File::directory?(path)
        path += "/"
      end
      # IN MERB, sta tutto in $context
      merb = MERB.new
 
      if @debug_mode then
        puts "input file name = #{input_file_name}"
        merb.convert( input_file_name, output_file_name ) ;
        #File.open(input_file_name, 'r') { |f| merb.template = f.read }
        #puts "SOURCE\n\n"
        #puts merb.source
        #puts "EVAL #{input_file_name}\n\n"
        #eval(merb.source)
        #puts MERB.out
        puts "output file name = #{output_file_name}"
      else
        merb.convert( input_file_name, output_file_name ) ;
      end
    end
  end

  if Dir.exists? "user-code" then
    Dir.find("./user-code", /\.(c|h).*$/) do |input_file_name|
      input = File.open(input_file_name, 'r' ) { |f| f.read() }
      output_file_name = @outputDir + '/ocp-cpp/srcs/' + input_file_name[12..-1] # elimino user-code
      File.unlink(output_file_name) if File.exist? output_file_name
      File.open(output_file_name, 'w') { |f| f.write(input)  }
    end
  end    
  #print "ALL DONE\n"
end

def read_parameters()

  @maxlenBC  = COMMON_Build::DATA[:Bc]                  . collect { |item| item[:name].length() }.max ;
  @maxlenMP  = COMMON_Build::DATA[:ModelParameters]     . collect { |item| item[:name].length() }.max ;
  @maxlenAUX = COMMON_Build::DATA[:AuxiliaryParameters] . collect { |item| item[:name].length() }.max ;
  @maxlenU   = COMMON_Build::DATA[:ConstraintU]         . collect { |item| item[:name].length() }.max ;
  @maxlen1D  = COMMON_Build::DATA[:Constraint1D]        . collect { |item| item[:name].length() }.max ;
  @maxlen2D  = COMMON_Build::DATA[:Constraint2D]        . collect { |item| item[:name].length() }.max ;

  @maxlenBC  = 0 unless @maxlenBC
  @maxlenMP  = 0 unless @maxlenMP
  @maxlenAUX = 0 unless @maxlenAUX
  @maxlenU   = 0 unless @maxlenU 
  @maxlen1D  = 0 unless @maxlen1D
  @maxlen2D  = 0 unless @maxlen2D


  # cerca le librerie da agganciare in atomatico e rimuove eventuale prefisso
  @meshInstance      = nil
  @meshPointer       = nil
  @UserClassInstance = []
  @UserClassPointer  = []
  @userHeaders       = {}
  @userNamespaces    = {}
  @userDataFile      = {}
  @lenicl            = 0 ;
  @lencl             = 0 ;
  COMMON_Build::DATA[:UserFunctionsClassInstances].each { |item|

    @userHeaders[item[:header]]       = 0 ; # simulo un insieme usando le chiavi di una hash
    @userNamespaces[item[:namespace]] = 0 ;
    @userDataFile[item[:setup_file]]  = 0 if item[:setup_file].kind_of? String ;

    cl_full      = item[:class]
    cl           = item[:class].sub(/(^[^#]+#)?/,'')  # rimuovo prefisso
    icl          = item[:instance]
    isMeshObject = item[:is_mesh_object] == "true"
    # controllo se la istanza è di tipo puntatore o meno
    # se è un puntatore allora vuole dire che viene definita dall'utente esternamente
    # e viene passato il puntatore, altrimenti è una classe contenuta nel modello.
    is_ptr = icl[0..0] == "*" ;
    if is_ptr then
      icl_p = icl[1..-1]
      icl   = icl[2..2].downcase + icl[3..-1] # nome senza "*p" con prima lettera minuscola
      @UserClassPointer  << { :class => cl, :class_full => cl_full, :instance => icl, :instance_p => icl_p, :is_mesh_object => isMeshObject }
    else
      @UserClassInstance << { :class => cl, :class_full => cl_full, :instance => icl, :is_mesh_object => isMeshObject }
    end
    @lencl  = cl.length  if  cl.length > @lencl
    @lenicl = icl.length if icl.length > @lenicl

    if isMeshObject then
      if @meshInstance or @meshPointer then
        print  "!!!!!!!!!!!!!!!!!!! ATTENTION !!!!!!!!!!!!!!!!!!!!!!!!!\n"+
               "Two mesh object found!!!\n"+
               "!!!!!!!!!!!!!!!!!!! ATTENTION !!!!!!!!!!!!!!!!!!!!!!!!!\n"
      end
      if is_ptr then
        @meshPointer = { :class => cl, :class_full => cl_full, :instance => icl, :instance_p => icl_p }
      else
        @meshInstance = { :class => cl, :class_full => cl_full, :instance => icl }
      end
    end
  }

  # se c'è un mesh object 
  if @meshPointer then
    tmp  = @meshPointer[:class]+'::SegmentClass const & segment'
    pist = @meshPointer[:instance_p]
    @meshObject = {
      :getInstance        => pist+'->',
      :getSegmentByIndex  => tmp +      ' = '+pist+'->getSegmentByIndex(i_segment) ;',
      :getSegmentByIndexL => tmp + 'Left  = '+pist+'->getSegmentByIndex(i_segment_left) ;',
      :getSegmentByIndexR => tmp + 'Right = '+pist+'->getSegmentByIndex(i_segment_right) ;'
    };
  elsif @meshInstance then
    tmp = @meshInstance[:class]+'::SegmentClass const & segment'
    ist = @meshInstance[:instance]
    @meshObject = {
      :getInstance        => ist + '.',
      :getSegmentByIndex  => tmp +      ' = '+ist+'.getSegmentByIndex(i_segment) ;',
      :getSegmentByIndexL => tmp + 'Left  = '+ist+'.getSegmentByIndex(i_segment_left) ;',
      :getSegmentByIndexR => tmp + 'Right = '+ist+'.getSegmentByIndex(i_segment_right) ;'
    };
  else
    print  "!!!!!!!!!!!!!!!!!!! ATTENTION !!!!!!!!!!!!!!!!!!!!!!!!!\n"+
           "No mesh object found!!!\n"+
           "!!!!!!!!!!!!!!!!!!! ATTENTION !!!!!!!!!!!!!!!!!!!!!!!!!\n"
  end

end

def subs_numbers( lines )
  out = ""
  lines.each do |line|
    ##line.gsub!( /\_\_\_i\_segment\_\_\_\#/,"getSegmentByIndex(i_segment).") cambiato con linea sucessiva
    line.gsub!( /unknown/,"result") # workaround per codegen
    line.gsub!( /\_\_\_i\_segment\_\_\_\#/,"segment.")
    line.gsub!( /\_\_\_i\_segment\_left\_\_\_\#/,"segmentLeft.")
    line.gsub!( /\_\_\_i\_segment\_right\_\_\_\#/,"segmentRight.")
    line.gsub!( /\_\_\_dot\_\_\_/,".")
    line.gsub!( /\_\_\_arrow\_\_\_/," -> ")
    line.gsub!( /\_\_\_void_argument\_\_\_/," ")
    line.gsub!( /operator\_\_\_pars\_\_\_\_/,"D_")
    line.gsub!( /operator\_\_\_pars\_\_\_/,"operator()")
    line.gsub!( /valueType\s+\_\_dummy\_\_\d+\s*=\s*(\w+)\_\_\_update\_\_\_(\w+)/ ) { |s| "#{$1}.update_#{$2}" }
    ## elimino assegnazioni __dummy__NUM = 
    line.gsub!( /valueType\s+\_\_dummy\_\_\d+\s*=\s*(.*)/ ) { |s| "#{$1}" }
    line.gsub!( /\b(Z|ZM|ZL|ZR)\[(\d+)\]/  ) { |s| "#{$1}[iZ_#{COMMON_Build::DATA[:Zvars][$2.to_i]}]" }
    line.gsub!( /\b(Q|QM|QL|QR)\[(\d+)\]/  ) { |s| "#{$1}[iQ_#{COMMON_Build::DATA[:Qvars][$2.to_i]}]" }
    line.gsub!( /\b(U|UM|UL|UR)\[(\d+)\]/  ) { |s| "#{$1}[iU_#{COMMON_Build::DATA[:Uvars][$2.to_i]}]" }
    line.gsub!( /\bP\[(\d+)\]/ )             { |s| "P[iP_#{COMMON_Build::DATA[:Pvars][$1.to_i]}]" }
    line.gsub!( /\bModelPars\[(\d+)\]/)      { |s| "ModelPars[iM_#{COMMON_Build::DATA[:ModelParameters][$1.to_i][:name]}]" }
    line.gsub!( /\bpow\(\w*([^,]+),\w*2\)/ )    { |s| "power2(#{$1})" }
    line.gsub!( /\bpow\(\w*([^,]+),\w*3\)/ )    { |s| "power3(#{$1})" }
    line.gsub!( /\bpow\(\w*([^,]+),\w*0\.2e1\)/ )   { |s| "power2(#{$1})" }
    line.gsub!( /\bpow\(\w*([^,]+),\w*0\.3e1\)/ )   { |s| "power3(#{$1})" }
    line.gsub!( /\bpow\(\w*([^,]+),\w+\-0\.2e1\)/ ) { |s| "rpower2(#{$1})" }
    line.gsub!( /\bpow\(\w*([^,]+),\w+\-0\.3e1\)/ ) { |s| "rpower3(#{$1})" }
    line.gsub!( /\b0\.(\d)e1\b/ ) { |s| "#{$1}.0" }
    out += line+"\n"
  end
  return out
end

def using_headers( indent = "" )
  tmp  = indent + "// Mechatronix Headers\n"
  tmp += indent + "#include <MechatronixCore/MechatronixCore.hh>\n"
  tmp += indent + "#include <MechatronixAstro/MechatronixAstro.hh>\n"
  tmp += indent + "#include <MechatronixSolver/MechatronixSolver.hh>\n"
  tmp += indent + "#include <MechatronixRoad/MechatronixRoad.hh>\n" if @use_road
  tmp += indent + "#include <MechatronixManufacturing/MechatronixManufacturing.hh>\n" if @use_toolPath2D
  tmp += indent + "#include <MechatronixInterfaceMruby/MechatronixInterfaceMruby.hh>\n"
  tmp += indent + "#include <MechatronixInterfaceLua/MechatronixInterfaceLua.hh>\n"
  if @userHeaders.length > 0 then
    tmp += "\n// user headers\n"
    @userHeaders.keys.each { |k| tmp += indent + k + "\n" }
  end
  return tmp ;
end

def using_namespaces( indent = "" )
  tmp  = indent + "using namespace std ;\n"
  tmp += indent + "// Mechatronix namespaces\n"
  tmp += indent + "using namespace MechatronixCore ;\n"
  tmp += indent + "using namespace MechatronixAstro ;\n"
  tmp += indent + "using namespace MechatronixSolver ;\n"
  tmp += indent + "using namespace MechatronixRoad ;\n"          if @use_road
  tmp += indent + "using namespace MechatronixManufacturing ;\n" if @use_toolPath2D
  tmp += indent + "using namespace MechatronixInterfaceMruby ;\n"
  tmp += indent + "using namespace MechatronixInterfaceLua ;\n"
  tmp += "\n"
  if @userNamespaces.length > 0 then
    tmp += indent + "// user namespaces\n"
    @userNamespaces.keys.each { |ns| tmp += indent + "using namespace #{ns} ;\n" }
  end
  return tmp ;
end
