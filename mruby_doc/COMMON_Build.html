def initialize( args, inDir, outDir )
  @inputDir  = inDir
  @outputDir = outDir
  @modelName = args.shift

  # defaults:
  @use_lua                     = false
  @controls_iterative          = false
  @debug_mode                  = false
  @exclude                     = []
  @external_libs_include_paths = []
  @external_libs_paths         = []
  @external_libs               = []

  args.each do |arg|
    case arg
    when /^--use_lua$/ then
      @use_lua = true
    when /^--controls_iterative$/ then
      @controls_iterative = true
    when /^--output_directory/ then
      if m = arg.match(/^--output_directory=(.*)$/) then
        @outputDir = m[1]
        @outputDir+= "/" if @outputDir[-2..-1] == ".."
      else
        raise ArgumentError, "Argument #{arg} bad syntax"
      end
    when /^--exclude/ then
      if m = arg.match(/^--exclude=\[(.*)\]$/) then
        m[1].split(',').each { |f| @exclude << f }
      else
        raise ArgumentError, "Argument #{arg} bad syntax"
      end
    when /^--debug/ then
      @debug_mode = true
    when /^--libs/ then
      if m = arg.match(/^--libs=\[(.*)\]$/) then
        m[1].split(',').each { |f| @external_libs << f }
      else
        raise ArgumentError, "Argument #{arg} bad syntax"
      end
    when /^--paths/ then
      if m = arg.match(/^--paths=\[(.*)\]$/) then
        m[1].split(',').each { |f| @external_libs_paths << f }
      else
        raise ArgumentError, "Argument #{arg} bad syntax"
      end
    when /^--include/ then
      if m = arg.match(/^--include=\[(.*)\]$/) then
        m[1].split(',').each { |f| @external_libs_include_paths << f }
      else
        raise ArgumentError, "Argument #{arg} bad syntax"
      end
    else
      raise ArgumentError, "Wrong argument #{arg}"
    end
  end

  @version  = "1.0"
  @time     = Time.now
  @dateFile = @time.day.to_s + '/' + @time.month.to_s + '/' + @time.year.to_s
end

def [](k)
  name = "@#{k.to_s}"
  if self.instance_variables.include?(name) then
    return self.instance_variable_get(name)
  elsif self.instance_variables.include?(name.to_sym) then # x mruby
    return self.instance_variable_get(name.to_sym)
  elsif DATA.key?(k.to_s)
    return DATA[k.to_s]
  elsif DATA.key?(k.to_sym)
    return DATA[k.to_sym]
  else
    print "\n\nMissing #{k}\nPossibles keys (only DATA)\n#{DATA.keys.sort}\n\n\n"
    return "#{k} not found"
  end
end

def constraints
  tmp = "  data.Constraints = {}\n"
  c1d = DATA[:Constraint1D]
  if c1d and c1d.length > 0 then
    tmp += "  # Constraint1D\n"
    tmp += "  # Penalty subtype: PENALTY_REGULAR, PENALTY_SMOOTH, PENALTY_PIECEWISE\n"
    tmp += "  # Barrier subtype: BARRIER_LOG, BARRIER_LOG_EXP, BARRIER_LOG0\n"
    len  = @maxlen1D+9
    c1d.each do |item|
      name = item[:name]
      cl   = item[:class]
      tmp += "  # #{cl}\n"
      tmp += "  data.Constraints[:#{name}] = {\n"
      case cl
      when /.*1DInterval/ then
        tmp += "    :subType   => #{item[:subtype]},\n"
        tmp += "    :epsilon   => #{parse_expr_ruby(item[:epsilon].to_s)},\n"
        tmp += "    :tolerance => #{parse_expr_ruby(item[:tolerance].to_s)},\n"
        tmp += "    :min       => #{parse_expr_ruby(item[:min].to_s)},\n"
        tmp += "    :max       => #{parse_expr_ruby(item[:max].to_s)},\n"
      when /Penalty1DPower/ then
        tmp += "    :exponent  => #{parse_expr_ruby(item[:exponent].to_s)},\n"
      when /Penalty1DBipower/ then
        tmp += "    :exponent  => #{parse_expr_ruby(item[:exponent].to_s)},\n"
        tmp += "    :epsilon   => #{parse_expr_ruby(item[:epsilon].to_s)},\n"
      else
        tmp += "    :subType   => #{item[:subtype]},\n"
        tmp += "    :epsilon   => #{parse_expr_ruby(item[:epsilon].to_s)},\n"
        tmp += "    :tolerance => #{parse_expr_ruby(item[:tolerance].to_s)},\n"
      end
      tmp += "    :active    => #{item[:active]}\n"
      tmp += "  }\n"
    end
  else
    tmp += "  # Constraint1D: none defined\n"
  end

  c2d = DATA[:Constraint2D]
  if c2d and c2d.length > 0 then
    tmp += "\n  # Constraint2D\n"
    len  = @maxlen2D+9
    c2d.each do |item|
      name = item[:name]
      cl   = item[:class]
      tmp += "  # #{cl}\n"
      tmp += "  data.Constraints[:#{name}] = {\n"
      tmp += "    :subType   => #{item[:subtype]},\n"
      tmp += "    :epsilon   => #{parse_expr_ruby(item[:epsilon].to_s)},\n"
      tmp += "    :tolerance => #{parse_expr_ruby(item[:tolerance].to_s)},\n"
      case cl
      when /(Penalty2DEllipse)|(Penalty2DRhomb)/ then
        tmp += "    :Xlimit   => #{parse_expr_ruby(item[:xlimit].to_s)},\n"
        tmp += "    :Ylimit   => #{parse_expr_ruby(item[:ylimit].to_s)},\n"

      tmp += "    :active    => #{item[:active]}\n"
      tmp += "  }\n"

  end
  return tmp

def controls
  ctrl = DATA[:ConstraintU]
  if ctrl and ctrl.length > 0 then
    tmp  = "  # Controls\n"
    tmp += "  # Penalty type controls: U_QUADRATIC, U_QUADRATIC2, U_PARABOLA, U_CUBIC\n"
    tmp += "  # Barrier type controls: U_LOGARITHMIC, U_COS_LOGARITHMIC, U_HYPERBOLIC\n\n"
    tmp += "  data.Controls = {}\n"
    ctrl.each do |item|
      name = item[:name]
      tmp += "  data.Controls[:#{name}] = {\n"
      tmp += "    :type      => #{item[:type]},\n"
      tmp += "    :epsilon   => #{parse_expr_ruby(item[:epsilon].to_s)},\n"
      tmp += "    :tolerance => #{parse_expr_ruby(item[:tolerance].to_s)}\n"
      tmp += "  }\n\n"
    end
  else
    tmp  = "  # Controls: No penalties or barriers constraint defined\n"
    tmp += "  data.Controls = {}\n"
  end
  return tmp
end

def get_function( method,
                  outVar       = "__result",
                  idmap        = nil,
                  prefix       = "id_",
                  extra_indent = "" )
  @lista_c = {}

  #puts "method = #{method}\n"
  #puts "outVar = #{outVar}\n"
  #puts "idmap  = #{idmap}\n"
  # calcolo lunghezza massima stringhe
  namelen = 15
  if (idmap and idmap.length > 0) then
    namelen = idmap.collect { |item| item.length }.max
    namelen = namelen + prefix.length
  end

  file_name = "#{@inputDir }#{method}.generated"
  if File::exist? file_name
    out_strings = []
    File::open(file_name, 'r') do |file|
      file.each do |line|
        line.chomp!
        # cerca le variabili del tipo 's', 't' e 'MapleGenVar' con numero seguite da `=`
        #m = /^\s*(s\d+)\s*=(.*)/.match(line)
        #m = /^\s*(t\d+)\s*=(.*)/.match(line)           unless m
        #m = /^\s*(MapleGenVar\d+)\s*=(.*)/.match(line) unless m
        
        # cerca le variabili assegnate cio seguite da `=`
        m = /^\s*(\w+)\s*=(.*)/.match(line)
        
        # aggiunge `valueType` a meno che la variabile non sia gia' stata definita prima
        if m then
          if @lista_c[m[1]] then
            line = "              #{m[1].ljust(4)} = #{m[2]}"
          else
            @lista_c[m[1]] = true
            line = "    valueType #{m[1].ljust(4)} = #{m[2]}"
          end
        end

        # cerca assegnazioni del tipo result[..] o unknown[..]
        line.gsub!( /(result|unknown)\s*\[/, "__result[" ) ;
        m = /^\s*\_\_result\[(\d+)\]\s*=\s*(.*)/.match(line)
        if m then
          # se trovate cambia assegnazione con outVar[..]
          if idmap.nil? then
            line = "    __result[ "+m[1].ljust(3)+" ] = " + m[2]
          else
            line = "    __result[ "+(prefix+idmap[m[1].to_i]).ljust(namelen)+" ] = " + m[2]
          end
        end
        line.gsub!( /\b\_\_result\b/, "#{outVar}" )

        line = '    ' + line.chomp.gsub(/^\s/, '')

        m = /\s*valueType\s+res\s*=\s*(.*)/.match(line.chomp)
        line = "    return " + m[1] if m
        out_strings << (extra_indent+line)
      end
    end
    return subs_numbers( out_strings )
  else
    # puts "Warning: cannot find file: #{file_name}\n"
    return "Warning: cannot find file: #{file_name}\n"
  end
end

 |      email: enrico.bertolazzi@unitn.it                                   |
 |             francesco.biral@unitn.it                                     |
 |             paolo.bosetti@unitn.it                                       |
  EOS
  case language
  when :c
    top    = "/*#{'-'*74}*\\\n"
    bottom = "\\*#{'-'*74}*/\n"
  when :ruby
    top = bottom = " \##{'-'*74}\#\n"
    text.gsub!(/\|/, '#')
  when :matlab

    text.gsub!(/\|/, '%')
  end
  return top+text+bottom
end

def to_string( data, indent = "  " )
  tmp     = ""
  indent2 = indent + "  "
  if data.kind_of? String then
    tmp = indent + parse_expr_ruby(data) + ",\n" ;
  elsif data.kind_of? Fixnum    or
        data.kind_of? Float     or
        data.kind_of? TrueClass or
        data.kind_of? FalseClass then
    tmp = indent + data.to_s + ",\n" ;
  elsif data.kind_of? Hash then
    lmax = data.keys.collect { |v| v.to_s.length }.max
    tmp  = "\n" + indent + "{\n" ;
    data.each do |key, value|
      key1 = key.to_s.ljust(lmax) ;
      tmp += indent2 + ":" + key1 + " => " + to_string(value,indent2).lstrip ;        
    end
    tmp += indent + "},\n" ;
  elsif data.kind_of? Array then
    tmp = "\n" + indent + "[\n" ;
    data.each do |value|
      tmp += indent2 + to_string(value,indent2) ;
    end
    tmp += indent + "],\n" ;      
  end
  return tmp
end
