def self.FunctionClassGenerator( description )
  $MECHATRONIX = Mechatronix::FunctionClassGeneratorForMerb.new description
  
  fname = $MECHATRONIX.fname

  $MECHATRONIX.clean_for_maple

  # costruisco file di comandi MAPLE
  $MECHATRONIX.template = $MECHATRONIX_FUNCTION_CLASS_MAPLET
  $MECHATRONIX.analyze
  $MECHATRONIX.write_to "#{fname}.maplet"
  ####$MECHATRONIX.convert("Mechatronix-function_class.maplet.tmpl","#{fname}.maplet")

  IO.popen("maple #{fname}.maplet") { |io| io.read } ##### TO BE SANED!
  #Kernel.`"maple #{fname}.maplet" #` per colorazione sintassi

  $MECHATRONIX.load_bodies

  $MECHATRONIX.template = $MECHATRONIX_FUNCTION_CLASS_HH
  $MECHATRONIX.analyze
  $MECHATRONIX.write_to "#{fname}.hh"

  $MECHATRONIX.template = $MECHATRONIX_FUNCTION_CLASS_CC
  $MECHATRONIX.analyze
  $MECHATRONIX.write_to "#{fname}.cc"
