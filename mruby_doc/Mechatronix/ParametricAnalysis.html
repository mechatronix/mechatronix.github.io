Overview

Simplifies the task of performing parametric analysis on an OCP or DS problem, providing a high-level interface.

Examples:

Run a parametric analysis:

pa = Mechatronix::ParametricAnalysis.new
pa.solver = ocp
pa.parameters = {:alpha => [0.2, 0.3, 0.4], :beta => [0.13, 0.14, 0.15]}

i = 0
pa.each do |solver, combination|
  desc =  "alpha = #{solver.data.Parameters[:alpha]}, beta = #{solver.data.Parameters[:beta]}"
  solver.solve
  solver.write_ocp_solution("data/#{problem_name}_OCP_result_#{i}.txt", desc)
  i += 1
end

def self.from_stdin
  payload = ""
  while line = STDIN.gets do
    payload << line
  end
  return YAML.load payload
end

def each
  raise ArgumentError, "Need a block!" unless block_given?
  unless @solver.kind_of?(OCPSolver) || @solver.kind_of?(DSSolver) then
    raise RuntimeError, "Set solver to a valid OCPSolver or DSSolver, first!" 
  end
  
  combine(@parameters) do |combination|
    if @multi_process then
      raise ArgumentError, "Multi-process not yet supported"
    else
      @parameters.keys.each_with_index do |k,i|
        @solver.data.Parameters[k] = combination[i]
      end
      @solver.setup
      yield @solver, combination
    end
  end
end
