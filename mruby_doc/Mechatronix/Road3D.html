def initialize( hsh = {} )
  super("road3d")
  self.Segments     = []
  self.MeshGridSize = hsh[:mesh_grid_size] || hsh[:MeshGridSize] || 1
  self.RoadWidth    = hsh[:road_width]     || hsh[:RoadWidth]    || 6
  
  self.S0       = hsh[:s]         || hsh[:S0]         || 0 # initial curvilinear abscissa\n"
  self.Theta0   = hsh[:theta]     || hsh[:Theta0]     || 0 # initial orientation in x-y plane\n"
  self.X0       = hsh[:x]         || hsh[:X0]         || 0 # initial x position of road middle line\n"
  self.Y0       = hsh[:y]         || hsh[:Y0]         || 0 # initial y position of road middle line\n"
  self.Z0       = hsh[:elevation] || hsh[:Elevation0] || 0 # initial z position of road middle line\n"
  self.Banking0 = hsh[:banking]   || hsh[:Banking0]   || 0
  self.IsSAE    = false ;
  if hsh[:IsSAE] || hsh[:is_sae] then
     self.Theta0 = - self.Theta0
     self.Y0     = - self.Y0
     self.Z0     = - self.Z0
     self.IsSAE  = true
  end
  self.Verbose = false
  self.Verbose = true if hsh[:Verbose] || hsh[:verbose] 
end

def add_segment_ISO( hsh )
  raise ArgumentError, "Need a kind of Hash" unless hsh.respond_to? :to_hash
  hsh = hsh.to_hash
  if hsh[:Radius] then
    curvature = 1.0/hsh[:radius]
    if hsh[:Angle] then
      length = (hsh[:Angle] / hsh[:Curvature] ).abs
    elsif hsh[:AngleDeg] then
      length = ((Math::PI*hsh[:AngleDeg]/180.0) / hsh[:Curvature] ).abs
    else
      raise RuntimeError, "Missing Angle field in Road segment #{self.Segments.length}"
    end
    initialCurvature = finalCurvature = curvature
  else
    initialCurvature = hsh[:InitialCurvature] || hsh[:Curvature] || 0 ;
    finalCurvature   = hsh[:FinalCurvature]   || hsh[:Curvature] || 0 ;
    raise RuntimeError, "Missing Length field in Road segment #{self.Segments.length}" unless hsh[:Length]
    length           = hsh[:Length]
  end
  self.Segments << {
    Length:            length,
    GridSize:          hsh[:GridSize]  ||
                       hsh[:grid_size] ||
                       self.MeshGridSize,
    Width:             hsh[:Width] || 
                       hsh[:width] ||
                       self.RoadWidth,
    InitialCurvature:  initialCurvature,
    FinalCurvature:    finalCurvature,
    InitialLeftWidth:  hsh[:initial_left_width] || 
                       hsh[:InitialLeftWidth]   ||
                       hsh[:left_width]         ||
                       hsh[:LeftWidth]          ||
                       self.RoadWidth / 2.0,
    FinalLeftWidth:    hsh[:FinalLeftWidth]     || 
                       hsh[:final_left_width]   || 
                       hsh[:LeftWidth]          || 
                       hsh[:left_width]         || 
                       self.RoadWidth / 2.0,
    InitialRightWidth: hsh[:InitialRightWidth]   || 
                       hsh[:initial_right_width] || 
                       hsh[:RightWidth]          ||
                       hsh[:right_width]         ||
                       self.RoadWidth / 2.0,
    FinalRightWidth:   hsh[:FinalRightWidth]     ||
                       hsh[:final_right_width]   ||
                       hsh[:RightWidth]          ||
                       hsh[:right_width]         ||
                       self.RoadWidth / 2.0,
    SpeedLimit:        hsh[:SpeedLimit]  ||
                       hsh[:speed_limit] ||
                       1000,
    Adherence:         hsh[:FrictionReductionFactor]   || 
                       hsh[:friction_reduction_factor] || 
                       hsh[:adherence]                 ||
                       1,
    FinalElevation:    hsh[:EndElevation]  ||
                       hsh[:end_elevation] ||
                       hsh[:elevation]     ||
                       0,
    FinalBanking:      hsh[:EndBanking]  ||
                       hsh[:end_banking] ||
                       hsh[:banking]     ||
                       0
  }
end

def add_segment_SAE( hsh )
  add_segment_ISO( hsh ) ;
  
  tmp = self.Segments[-1]
  tmp[:InitialCurvature] = - tmp[:InitialCurvature]
  tmp[:FinalCurvature]   = - tmp[:FinalCurvature]
  tmp[:FinalElevation]   = - tmp[:FinalElevation]
  tmp[:FinalBanking]     = - tmp[:FinalBanking]
  self.Segments[-1] = tmp

end
