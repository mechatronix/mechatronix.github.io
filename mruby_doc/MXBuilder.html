Overview

This class provides a simplified build environment for compiling and linking into a dynamic library the files generated by Maple and corresponding to an OCP or DS problem. The resulting dynamic library can be later loaded in mruby by using the Mechatronix::OCPSolver.new method.

Instance parameters are automatically initialized to sensible values for Mac, Linux, and Windows build environment (VisualStudio 2010 on Windows)

Examples:

Typical call sequence:

if ! File.exist?("lib/lib#{problem_name}.dylib") then
  require "build"
  compiler = MXBuilder.new(problem_name)
  compiler.build
end

def initialize(name, src_base = 'ocp-cpp')
  @name = name
  @src_base = src_base
  case RUBY_PLATFORM
  when /darwin/
    @l_flags = "-arch x86_64 -dynamiclib -current_version 1.0 -std=c++11 -stdlib=libc++"
    @c_flags = "-msse4.2 -msse4.1 -mssse3 -msse3 -msse2 -msse -mmmx -m64 -O3 -funroll-loops -fPIC -std=c++11 -stdlib=libc++ -arch x86_64"
    @libs = "-F/Library/Frameworks -framework Accelerate -framework MechatronixCore -framework MechatronixAstro -framework MechatronixInterfaceLua -framework MechatronixInterfaceMruby -framework MechatronixODE -framework MechatronixRoad -framework MechatronixSolver -framework MechatronixVehicle -framework MechatronixManufacturing"
    @includes = "-I/usr/local/include -I./#{@src_base}/srcs -F/Library/Frameworks"
    @c = "clang++"
    @l = "clang++"
    @obj_ext = ".o"
    @obj_match = /\.o$/
  when /mswin/
    @c = "cl.exe"
    @c_flags = "/nologo /MD /Zi /W3 /WX- /EHa /GS /Od /D \"#{@name.upcase}_EXPORT\""
    @includes = "/IC:/Mechatronix/win32/include /I./#{@src_base}/srcs"
    @l = "link.exe"
    @l_flags = "/LINK /DLL"
    @libs = "/LIBPATH:C:/Mechatronix/win32/lib"
    @obj_ext = ".obj"
    @obj_match = /\.obj$/
  when /linux/
    @l_flags = "-L/usr/local/lib/ -L/usr/lib/  -L/usr/lib/atlas-base/ -L/usr/lib/atlas-base/atlas/ -stdlib=libstdc++ -lstdc++ -lpthread -lreadline -ldl -lpcre -llapack_atlas -llapack -lcblas -Wl,-rpath=. -Wl,-rpath=./lib -Wl,-rpath=/usr/lib/openblas-base -Wl,-rpath=/usr/lib/atlas-base"
    @c_flags = "-msse4.2 -msse4.1 -mssse3 -msse3 -msse2 -msse -mmmx -O3 -funroll-loops -fPIC -std=c++11"
    @libs = %w(MechatronixCore MechatronixAstro MechatronixSolver MechatronixInterfaceLua MechatronixInterfaceMruby MechatronixODE MechatronixRoad MechatronixVehicle MechatronixManufacturing).inject("-Wl,--whole-archive ") {|s,l| s + " -l#{l}"} + " -Wl,--no-whole-archive"
    @includes = "-I/usr/include -I/usr/include/atlas -I/usr/local/include -I./#{@src_base}/srcs"
    @c = "clang++"
    @l = "clang++"
    @obj_ext = ".o"
    @obj_match = /\.o$/
  else
    raise "Unsupported platform"
  end
  @src  = "#{@src_base}/srcs"
  @main = "#{@src_base}/#{@name}_Main.cc"
  @dest = "lib"
  
  @add_includepaths = ""
  @add_libs         = ""
  @add_libpaths     = ""
end

  end

def compile
  ext_match = /\.cc/
  Dir.foreach(@src) do |s|
    if s.match(ext_match) then
      dst = s.gsub(ext_match, @obj_ext)
      cmd = compile_str(s, dst)
      puts "CC #{s} -> #{dst}"
      out, result = shell(cmd)
      if (result != 0) then
        puts cmd
        puts out
        return false
      end
    end
  end
end

def dll_name
  case RUBY_PLATFORM

  end
end

def link
  objs = []
  Dir.mkdir "lib" unless Dir.exist? "lib"
  Dir.foreach(@src) do |obj|
    objs << "#{@src}/#{obj}" if obj.match(@obj_match)
  end
  cmd = link_str(objs)
  puts "LN lib/#{dll_name}"
  out, result = shell(cmd)
  if (result != 0) then
    puts cmd
    puts out
    return false
  end
  
  cmd = static_link_str(objs)
  puts "LN lib/#{lib_name}"
  out, result = shell(cmd)
  if (result != 0) then
    puts cmd
    puts out
    return false
  end
  return true
end
